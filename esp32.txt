#define PIN_GREEN 13

#define PIN_BUILTIN 2
#define PIN_DHT 15
#define DHTTYPE DHT11

// ============================================================

DHT dht(PIN_DHT, DHTTYPE);

// Traffic light state
bool systemOn = false;
bool builtinLedOn = false;

String activeMode = "idle";  // idle | auto | manual | builtin

const unsigned long DHT_INTERVAL = 5000;  // 5 seconds

// Last processed command ID to avoid re-processing
long lastCommandId = -1;

// ============================================================
// HELPERS: BUZZER
// ============================================================
void beep(int freq = 1000, int dur = 150) {
  tone(PIN_BUZZER, freq, dur);
  delay(dur + 20);
  noTone(PIN_BUZZER);
}

void alertBuzzer() {
  // Short-short beep for state change
  beep(1200, 100);
  delay(80);
  beep(1200, 100);
}

void warnBuzzer() {
  // Long beep for baby environment warning
  beep(800, 400);
  delay(100);
  beep(800, 400); 
}

// ============================================================
// HELPERS: LED CONTROL
// ============================================================
void allLedsOff() {
  digitalWrite(PIN_RED, LOW);
  digitalWrite(PIN_YELLOW, LOW);
  digitalWrite(PIN_GREEN, LOW);
}

void setLed(String color) {
  allLedsOff();
  if (color == "red") digitalWrite(PIN_RED, HIGH);
  else if (color == "yellow") digitalWrite(PIN_YELLOW, HIGH);
  else if (color == "green") digitalWrite(PIN_GREEN, HIGH);
  currentLed = color;
}

// ============================================================
// WIFI CONNECT
// ============================================================
void connectWiFi() {
  Serial.print("[WiFi] Connecting to ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n[WiFi] Connected: " + WiFi.localIP().toString());
  } else {
    Serial.println("\n[WiFi] Failed to connect");
  }
}

// ============================================================
// SUPABASE: POST device status
// ============================================================
void postDeviceStatus(String device, bool online) {
  if (WiFi.status() != WL_CONNECTED) return;
  HTTPClient http;
  String endpoint = String(supabaseUrl) + "/rest/v1/device_status";
  http.begin(endpoint);
  http.addHeader("Content-Type", "application/json");
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", String("Bearer ") + supabaseKey);
  http.addHeader("Prefer", "resolution=merge-duplicates");

  StaticJsonDocument<128> doc;
  doc["device"] = device;
  doc["online"] = online;
  doc["updated_at"] = "now()";
  String body;
  serializeJson(doc, body);

  int code = http.PUT(body);
  if (code < 0) {
    // Try INSERT if PUT fails (first time)
    http.end();
    http.begin(endpoint);
    http.addHeader("Content-Type", "application/json");
    http.addHeader("apikey", supabaseKey);
    http.addHeader("Authorization", String("Bearer ") + supabaseKey);
    http.POST(body);
  }
  http.end();
}

// ============================================================
// SUPABASE: INSERT sensor reading (DHT)
// ============================================================
void postSensorReading(float temp, float hum) {
  if (WiFi.status() != WL_CONNECTED) return;
  HTTPClient http;
  String endpoint = String(supabaseUrl) + "/rest/v1/sensor_readings";
  http.begin(endpoint);
  http.addHeader("Content-Type", "application/json");
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", String("Bearer ") + supabaseKey);

  StaticJsonDocument<128> doc;
  doc["temperature"] = temp;
  doc["humidity"] = hum;
  String body;
  serializeJson(doc, body);

  int code = http.POST(body);
  Serial.printf("[DHT] POST → %d  T=%.1f  H=%.0f%%\n", code, temp, hum);
  http.end();
}

// ============================================================
// SUPABASE: FETCH latest traffic command
// ============================================================
void fetchTrafficCommand() {
  if (WiFi.status() != WL_CONNECTED) return;
  HTTPClient http;

  // Get latest unprocessed command
  String endpoint = String(supabaseUrl)
                    + "/rest/v1/traffic_commands?select=id,command,value&order=id.desc&limit=1";
  http.begin(endpoint);
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", String("Bearer ") + supabaseKey);
  http.addHeader("Accept", "application/json");

  int code = http.GET();
  if (code == 200) {
    String payload = http.getString();
    DynamicJsonDocument doc(256);
    DeserializationError err = deserializeJson(doc, payload);
    if (!err && doc.is<JsonArray>() && doc.size() > 0) {
      long id = doc[0]["id"];
      String command = doc[0]["command"].as<String>();
      String value = doc[0]["value"].as<String>();

      if (id != lastCommandId) {
        lastCommandId = id;
        processCommand(command, value);
      }
    }
  }
  http.end();
}

// ============================================================
// PROCESS COMMAND FROM DASHBOARD
// ============================================================
void processCommand(String command, String value) {
  Serial.printf("[CMD] %s = %s\n", command.c_str(), value.c_str());

  if (command == "system") {
    if (value == "on") {
      systemOn = true;
      activeMode = "auto";
      Serial.println("[System] ON");
    } else {
      systemOn = false;
      activeMode = "idle";
      allLedsOff();
      digitalWrite(PIN_BUILTIN, LOW);
      Serial.println("[System] OFF");
    }
  }

  else if (command == "led") {
    if (!systemOn) return;
    if (value == "off") {
      allLedsOff();
      currentLed = "none";
      activeMode = "idle";
    } else {
      setLed(value);
      activeMode = "manual";
      alertBuzzer();
    }
  }

  else if (command == "builtin_led") {
    if (!systemOn) return;
    bool on = (value == "on");
    digitalWrite(PIN_BUILTIN, on ? HIGH : LOW);
    builtinLedOn = on;
    activeMode = on ? "builtin" : "idle";
  }

  else if (command == "buzzer") {
    if (!systemOn) return;
    if (value == "beep") alertBuzzer();
  }
}

// ============================================================
// READ DHT SENSOR
// ============================================================
void readDHT() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  if (isnan(h) || isnan(t)) {
    Serial.println("[DHT] Read failed");
    return;
  }

  Serial.printf("[DHT] Temp: %.1f°C  Humidity: %.0f%%\n", t, h);
  postSensorReading(t, h);

  // Check baby safety
  bool tempOk = (t >= 24.0 && t <= 27.0);
  bool humOk = (h >= 40.0 && h <= 60.0);

  if (!tempOk || !humOk) {
    Serial.println("[DHT] WARNING: Conditions not ideal for baby!");
    warnBuzzer();
  }
}

// ============================================================
// SETUP
// ============================================================
void setup() {
  Serial.begin(115200);
  Serial.println("\n[Boot] SmartIoT ESP32 Firmware");

  // Pin modes
  pinMode(PIN_RED, OUTPUT);
  pinMode(PIN_YELLOW, OUTPUT);
  pinMode(PIN_GREEN, OUTPUT);
  pinMode(PIN_BUZZER, OUTPUT);
  pinMode(PIN_BUILTIN, OUTPUT);
  allLedsOff();
  digitalWrite(PIN_BUILTIN, LOW);

  // DHT
  dht.begin();

  // WiFi
  connectWiFi();

  // Announce device online
  if (WiFi.status() == WL_CONNECTED) {
    postDeviceStatus("esp32-traffic", true);
    postDeviceStatus("esp32-dht", true);
    // Startup beep
    beep(1500, 100);
    delay(80);
    beep(1800, 100);
  }
}

// ============================================================
// LOOP
// ============================================================
void loop() {
  unsigned long now = millis();

  // Reconnect WiFi if dropped
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("[WiFi] Reconnecting…");
    WiFi.reconnect();
    delay(2000);
    return;
  }

  // Poll commands from Supabase dashboard
  if (now - lastCmdPoll >= POLL_INTERVAL) {
    lastCmdPoll = now;
    fetchTrafficCommand();
  }

  // Read DHT every 5 seconds
  if (now - lastDhtRead >= DHT_INTERVAL) {
    lastDhtRead = now;
    readDHT();
  }

  delay(50);  // Small yield
}
